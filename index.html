browser-shim by laurie71

Easy code sharing between browser and NodeJS. 
var shim = require('browser-shim');
var pkg = shim.provide('my.awesome.package');

pkg.greet = function(name) {
    console.log('Hello ' + name + '!');
};

var shim = require('browser-shim');
var pkg = require('shared-module');

shim.httpShim('/shim.js', 5040);
pkg.greet('Node');
// ...

<html>
<head>
    <title>Shim In!</title>
    <script src="http://localhost:5040/shim.js"></script>
    <script src="http://localhost:5040/shared-module.js"></script>
    <script>
        pkg.greet('Browser');
    </script>
</head>
<body>
    <h1>Shim In!</h1>
    <p>Check the browser console...</p>
</body>
</html>
